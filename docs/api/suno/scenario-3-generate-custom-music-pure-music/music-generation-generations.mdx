## Overview

Music generation interface, can generate new music directly or continue existing musicIf you want to continue a user-uploaded music, you need to add `-upload` after the mv field, for example `chirp-v3-5-upload`

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `Bearer YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="prompt" type="string" required default="">
  Prompt parameter
</ParamField>

<ParamField body="tags" type="string"  default="heavy metal">
  Tags parameter
</ParamField>

<ParamField body="mv" type="string"  default="chirp-v3-5">
  Mv parameter
</ParamField>

<ParamField body="title" type="string"  default="Starry ">
  Title parameter
</ParamField>

<ParamField body="continue_clip_id" type="string"  default="None">
  Continue Clip Id parameter
</ParamField>

<ParamField body="continue_at" type="string"  default="None">
  Continue At parameter
</ParamField>

<ParamField body="infill_start_s" type="string"  default="None">
  Infill Start S parameter
</ParamField>

<ParamField body="infill_end_s" type="string"  default="None">
  Infill End S parameter
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/suno/generate"

    payload := []byte(`{
"prompt": "",
"tags": "heavy metal",
"mv": "chirp-v3-5",
"title": "Starry ",
"continue_clip_id": null,
"continue_at": null,
"infill_start_s": null,
"infill_end_s": null
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
