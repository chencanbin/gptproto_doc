## Overview

This endpoint provides image to image functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `Bearer YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="messages" type="array" required default="[{&quot;role&quot;: &quot;user&quot;, &quot;content&quot;: [{&quot;type&quot;: &quot;text&quot;, &quot;text&quot;: &quot;Make this colorful&quot;}, {&quot;type&quot;: &quot;image_url&quot;, &quot;image_url&quot;: {&quot;url&quot;: &quot;https://pics.144096.xyz/images/logo1.png&quot;}}]}]">
  Array of message objects for the conversation
</ParamField>

<ParamField body="model" type="string" required default="gemini-2.5-flash-image-preview-hd">
  The model to use for the request
</ParamField>

<ParamField body="stream" type="boolean"  default="false">
  Whether to stream the response
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/chat/completions"

    payload := []byte(`{
"messages": [
{
"role": "user",
"content": [
{
"type": "text",
"text": "Make this colorful"
},
{
"type": "image_url",
"image_url": {
"url": "https://pics.144096.xyz/images/logo1.png"
}
}
]
}
],
"model": "gemini-2.5-flash-image-preview-hd",
"stream": false
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
