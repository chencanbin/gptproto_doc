## Authentication

This endpoint requires authentication using a Bearer token.

</ParamField>

## Request Body

<ParamField body="model" type="string" required default="claude-sonnet-4-thinking">
  The model to use for the request
</ParamField>

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/chat/completions"

    payload := []byte(`{
"model": "claude-sonnet-4-thinking",
"messages": [
{
"role": "user",
"content": [
{
"type": "text",
"text": "Summarize the contents of this document"
},
{
"type": "file",
"file": {
"filename": "api-doc.pdf",
"file_data": "https://www.bt.cn/data/api-doc.pdf"
}
}
]
}
],
"max_tokens": 1000,
"stream": false
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
