## Overview

This endpoint provides responses mode functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" type="string" required default="sk-***********">
  Your API key in the format: `Bearer YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="model" type="string" required default="gpt-4.1-2025-04-14">
  The model to use for the request
</ParamField>

<ParamField body="input" type="string" required default="Tell me a three sentence bedtime story about a unicorn.">
  Input parameter
</ParamField>

## Request Example

<CodeGroup>

```bash cURL
curl -X POST "https://gptproto.com/v1/responses" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
  "model": "gpt-4.1-2025-04-14",
  "input": "Tell me a three sentence bedtime story about a unicorn."
}'
```

```python Python
import requests
import json

url = "https://gptproto.com/v1/responses"
headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
}

data = {
  "model": "gpt-4.1-2025-04-14",
  "input": "Tell me a three sentence bedtime story about a unicorn."
}

response = requests.post(url, headers=headers, json=data)
result = response.json()
print(json.dumps(result, indent=2))
```

```javascript JavaScript
const url = "https://gptproto.com/v1/responses";
const headers = {
  "Authorization": "Bearer YOUR_API_KEY",
  "Content-Type": "application/json"
};

const data = {
  "model": "gpt-4.1-2025-04-14",
  "input": "Tell me a three sentence bedtime story about a unicorn."
};

fetch(url, {
  method: "POST",
  headers: headers,
  body: JSON.stringify(data)
})
  .then(response => response.json())
  .then(data => console.log(data))
  .catch(error => console.error("Error:", error));
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto.com/v1/responses"

    payload := []byte(`{
"model": "gpt-4.1-2025-04-14",
"input": "Tell me a three sentence bedtime story about a unicorn."
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
