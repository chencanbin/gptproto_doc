## Overview

This endpoint provides text to video functionality.

## Authentication

This endpoint requires authentication using a Bearer token.

<ParamField header="Authorization" default="sk-***********" type="string" required>
  Your API key in the format: `Bearer YOUR_API_KEY`
</ParamField>

## Request Body

<ParamField body="prompt" default="A video of a horse dancing." type="string" required>
  Prompt parameter
</ParamField>

<ParamField body="size" default="720*1280" type="string">
  Size parameter
</ParamField>

<ParamField body="duration" default="4" type="integer">
  Duration parameter
</ParamField>

## Request Example

<CodeGroup>

```bash cURL
curl -X POST "https://gptproto/api/v3/openai/sora-2/text-to-video" \
  -H "Authorization: Bearer YOUR_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
  "prompt": "A video of a horse dancing.",
  "size": "720*1280",
  "duration": 4
}'
```


```python Python
import requests
import json

url = "https://gptproto/api/v3/openai/sora-2/text-to-video"
headers = {
    "Authorization": "Bearer YOUR_API_KEY",
    "Content-Type": "application/json"
}

data = {
  "prompt": "A video of a horse dancing.",
  "size": "720*1280",
  "duration": 4
}

response = requests.post(url, headers=headers, json=data)
result = response.json()
print(json.dumps(result, indent=2))
```


```javascript JavaScript
const url = "https://gptproto/api/v3/openai/sora-2/text-to-video";
const headers = {
  "Authorization": "Bearer YOUR_API_KEY",
  "Content-Type": "application/json"
};

  "prompt": "A video of a horse dancing.",

fetch(url, {
  method: "POST",
  headers: headers,
  body: JSON.stringify(data)

  .catch(error => console.error("Error:", error));
```


```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io/ioutil"
    "net/http"
)

func main() {
    url := "https://gptproto/api/v3/openai/sora-2/text-to-video"

    payload := []byte(`{
"prompt": "A video of a horse dancing.",
"size": "720*1280",
"duration": 4
}`)

    req, _ := http.NewRequest("POST", url, bytes.NewBuffer(payload))
    req.Header.Set("Authorization", "Bearer YOUR_API_KEY")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        panic(err)
    }
    defer resp.Body.Close()

    body, _ := ioutil.ReadAll(resp.Body)
    fmt.Println(string(body))
}
```
